async function initializeApp() { try { setStatus("Loading schema and images..."); const [schemaResp, imagesResp] = await Promise.all([ fetch("/schema", { cache: "no-store" }), fetch("/images-list", { cache: "no-store" }), ]); if (!schemaResp.ok) throw new Error(`Failed loading schema: ${schemaResp.status}`); if (!imagesResp.ok) throw new Error(`Failed loading images: ${imagesResp.status}`); schema = await schemaResp.json(); images = await imagesResp.json(); renderQuestions(); renderImage(); setStatus("Ready"); } catch (error) { setStatus(error.message || "Initialization failed", "error"); submitBtn.disabled = true; } } enterBtn.addEventListener("click", () => { const id = annotatorInput.value.trim(); if (!id) { setStatus("Enter annotator ID first", "error"); return; } setStatus("Annotator ID set", "ok"); }); downloadBtn.addEventListener("click", async () => { const token = annotatorInput.value.trim(); if (!token) { setStatus("Enter admin token first", "error"); return; } try { window.open(`/admin/export/annotations?token=${encodeURIComponent(token)}`, "_blank"); setStatus("CSV download started", "ok"); } catch (error) { setStatus(error.message || "Export failed", "error"); } }); submitBtn.addEventListener("click", async () => { if (!images.length || idx >= images.length) { setStatus("No images left to annotate", "error"); return; } const annotatorId = annotatorInput.value.trim(); if (!annotatorId) { setStatus("Annotator ID is required", "error"); return; } const { answers, missingSingles } = collectAnswers(); if (missingSingles.length) { setStatus(`Complete required single-choice questions: ${missingSingles.join(", ")}`, "error"); return; } submitBtn.disabled = true; setStatus("Submitting..."); try { const response = await fetch("/annotate", { method: "POST", headers: { "Content-Type": "application/json" }, body: JSON.stringify({ image_id: images[idx], annotator_id: annotatorId, answers, }), }); if (!response.ok) { const payload = await response.text(); throw new Error(payload || `Submit failed: ${response.status}`); } idx += 1; clearForm(); renderImage(); if (idx >= images.length) { setStatus("All annotations submitted", "ok"); return; } setStatus("Saved", "ok"); } catch (error) { setStatus(error.message || "Submit failed", "error"); submitBtn.disabled = false; return; } if (idx < images.length) submitBtn.disabled = false; }); initializeApp();
